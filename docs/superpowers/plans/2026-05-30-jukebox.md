# Jukebox (Win98 media player) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an offline-first, Win98-styled audio player module ("Jukebox") to Ghost Access 98 — local library playback, embedded-tag metadata, M3U playlists, a WinAmp skin with spectrum visualizer, and opt-in internet streaming.

**Architecture:** Local files reach the renderer through a path-confined `ga98media://` custom protocol with HTTP range support (registered in main, validated against remembered library roots + a session allowlist). Metadata + library index persist as JSON under `dataRoot` via the existing secure-fs shim (vault-encrypted at rest). The renderer module follows the established 4-point registration pattern and plays audio via HTML5 `<audio>` + Web Audio `AnalyserNode`. Internet streaming is gated at the application layer by a single `media.streamingEnabled` setting (default off) — not CSP, because EyeSpy already shares the renderer's remote-media CSP.

**Tech Stack:** Electron 33, React 18, TypeScript 5.7 (strict), `music-metadata` (pure-JS tag parser, pending Stage 0 verification), `hls.js` (already a dependency), vitest, secure-fs.

**Spec:** `docs/superpowers/specs/2026-05-30-jukebox-design.md`

```ts
// src/main/media/protocol.ts
import { realpathSync } from 'node:fs';
import { sep } from 'node:path';

/** True iff `candidate` resolves (realpath, symlinks followed) to a file inside one of
 *  `roots`, OR its realpath is in the session `allowlist`. Fails closed on any error. */
export function isAuthorizedMediaPath(candidate: string, roots: string[], allowlist: Set<string>): boolean {
  let real: string;
  try { real = realpathSync(candidate); } catch { return false; }
  if (allowlist.has(real)) return true;
  for (const root of roots) {
    let realRoot: string;
    try { realRoot = realpathSync(root); } catch { continue; }
    const prefix = realRoot.endsWith(sep) ? realRoot : realRoot + sep;
    if (real === realRoot || real.startsWith(prefix)) return true;
  }
  return false;
}
```

```ts
// append to src/main/media/protocol.ts
import { createReadStream, statSync } from 'node:fs';
import { Readable } from 'node:stream';
import { protocol } from 'electron';
import { getLibraryRoots } from './library';

/** Realpaths the user authorized this session via Open files / Load playlist. */
export const adHocAllowlist = new Set<string>();

const MIME: Record<string, string> = {
  mp3: 'audio/mpeg', m4a: 'audio/mp4', aac: 'audio/aac', flac: 'audio/flac',
  wav: 'audio/wav', ogg: 'audio/ogg', oga: 'audio/ogg', opus: 'audio/ogg'
};
function mimeFor(p: string): string {
  const ext = p.slice(p.lastIndexOf('.') + 1).toLowerCase();
  return MIME[ext] ?? 'application/octet-stream';
}

/** Decode ga98media://track/<encodeURIComponent(absPath)> → absolute path. */
function pathFromRequest(url: string): string | null {
  try { const u = new URL(url); return decodeURIComponent(u.pathname.replace(/^\//, '')); }
  catch { return null; }
}

export function registerMediaProtocol(): void {
  protocol.handle('ga98media', async (request) => {
    const p = pathFromRequest(request.url);
    if (!p) return new Response('bad request', { status: 400 });
    const roots = await getLibraryRoots();
    if (!isAuthorizedMediaPath(p, roots, adHocAllowlist)) {
      return new Response('forbidden', { status: 403 });
    }
    let size: number;
    try { size = statSync(p).size; } catch { return new Response('not found', { status: 404 }); }

    const range = request.headers.get('range');
    const m = range && /^bytes=(\d*)-(\d*)$/.exec(range);
    if (m) {
      const start = m[1] ? parseInt(m[1], 10) : 0;
      const end = m[2] ? parseInt(m[2], 10) : size - 1;
      if (Number.isNaN(start) || start > end || end >= size) {
        return new Response('range not satisfiable', { status: 416, headers: { 'Content-Range': `bytes */${size}` } });
      }
      const stream = createReadStream(p, { start, end });
      return new Response(Readable.toWeb(stream) as ReadableStream, {
        status: 206,
        headers: {
          'Content-Type': mimeFor(p),
          'Content-Length': String(end - start + 1),
          'Content-Range': `bytes ${start}-${end}/${size}`,
          'Accept-Ranges': 'bytes'
        }
      });
    }
    const stream = createReadStream(p);
    return new Response(Readable.toWeb(stream) as ReadableStream, {
      status: 200,
      headers: { 'Content-Type': mimeFor(p), 'Content-Length': String(size), 'Accept-Ranges': 'bytes' }
    });
  });
}
```

```ts
// src/main/media/library.ts
import { randomUUID } from 'node:crypto';
import { join } from 'node:path';
import { app } from 'electron';
import { secureReadText, secureWriteText } from '../storage/secure-fs';
import type { MediaLibrarySnapshot, MediaStation, MediaTrack } from '@shared/post-mvp-types';

const file = (): string => join(app.getPath('userData'), 'media-library.json');
const EMPTY: MediaLibrarySnapshot = { roots: [], tracks: [], stations: [] };

async function read(): Promise<MediaLibrarySnapshot> {
  try { return { ...EMPTY, ...JSON.parse(await secureReadText(file())) }; }
  catch { return { ...EMPTY }; }
}
async function write(s: MediaLibrarySnapshot): Promise<void> {
  await secureWriteText(file(), JSON.stringify(s, null, 2));
}

export async function _resetForTest(): Promise<void> { await write({ ...EMPTY }); }
export async function getSnapshot(): Promise<MediaLibrarySnapshot> { return read(); }
export async function getLibraryRoots(): Promise<string[]> { return (await read()).roots; }

export async function addRoot(root: string): Promise<MediaLibrarySnapshot> {
  const s = await read();
  if (!s.roots.includes(root)) s.roots.push(root);
  await write(s); return s;
}
export async function removeRoot(root: string): Promise<MediaLibrarySnapshot> {
  const s = await read();
  s.roots = s.roots.filter((r) => r !== root);
  s.tracks = s.tracks.filter((t) => !t.path.startsWith(root));
  await write(s); return s;
}
export async function upsertStation(input: { id?: string; label: string; url: string }): Promise<MediaStation> {
  const s = await read();
  const station: MediaStation = { id: input.id ?? randomUUID(), label: input.label, url: input.url };
  const i = s.stations.findIndex((x) => x.id === station.id);
  if (i >= 0) s.stations[i] = station; else s.stations.push(station);
  await write(s); return station;
}
export async function deleteStation(id: string): Promise<void> {
  const s = await read(); s.stations = s.stations.filter((x) => x.id !== id); await write(s);
}
export async function setTracks(tracks: MediaTrack[]): Promise<void> {
  const s = await read(); s.tracks = tracks; await write(s);
}
```

```ts
// add to src/main/media/library.ts
import { readdir, stat, mkdir, writeFile } from 'node:fs/promises';
import { parseFile } from 'music-metadata';

const AUDIO_EXT = new Set(['mp3', 'm4a', 'aac', 'flac', 'wav', 'ogg', 'oga', 'opus']);
const artDir = (): string => join(app.getPath('userData'), 'media-art');

async function* walk(dir: string): AsyncGenerator<string> {
  let entries; try { entries = await readdir(dir, { withFileTypes: true }); } catch { return; }
  for (const e of entries) {
    const full = join(dir, e.name);
    if (e.isDirectory()) yield* walk(full);
    else if (AUDIO_EXT.has(e.name.slice(e.name.lastIndexOf('.') + 1).toLowerCase())) yield full;
  }
}

export async function refresh(): Promise<MediaLibrarySnapshot> {
  const s = await read();
  const prev = new Map(s.tracks.map((t) => [t.path, t]));
  const next: MediaTrack[] = [];
  for (const root of s.roots) {
    for await (const path of walk(root)) {
      const st = await stat(path);
      const old = prev.get(path);
      if (old && old.mtime === st.mtimeMs && old.size === st.size) { next.push(old); continue; }
      const track: MediaTrack = { path, mtime: st.mtimeMs, size: st.size };
      try {
        const meta = await parseFile(path, { duration: true });
        track.title = meta.common.title;
        track.artist = meta.common.artist;
        track.album = meta.common.album;
        track.durationMs = meta.format.duration ? Math.round(meta.format.duration * 1000) : undefined;
        const pic = meta.common.picture?.[0];
        if (pic) {
          await mkdir(artDir(), { recursive: true });
          const ref = `${st.mtimeMs.toString(36)}-${path.length}.img`;
          await writeFile(join(artDir(), ref), pic.data);
          track.artRef = ref;
        }
      } catch { /* unreadable tags: keep filename-only entry */ }
      next.push(track);
    }
  }
  s.tracks = next;
  await write(s);
  return s;
}
```

```ts
export function ensureMediaRoot(p: unknown): string {
  if (typeof p !== 'string' || p.length === 0 || p.length > 4096) throw new ValidationError('root must be a path string');

export function ensureStationInput(v: unknown): { id?: string; label: string; url: string } {
  if (!v || typeof v !== 'object') throw new ValidationError('station must be an object');
  const o = v as { id?: unknown; label?: unknown; url?: unknown };
  if (typeof o.label !== 'string' || o.label.trim().length === 0 || o.label.length > 120) {

  const url = validateExternalUrl(o.url); // enforces http/https, throws otherwise
  const out: { id?: string; label: string; url: string } = { label: o.label.trim(), url };
```

```tsx
import { useEffect, useRef } from 'react';
export function Visualizer({ analyser, enabled }: { analyser: AnalyserNode | null; enabled: boolean }): JSX.Element {
  const ref = useRef<HTMLCanvasElement>(null);
  useEffect(() => {
    if (!enabled || !analyser || !ref.current) return;
    const canvas = ref.current, ctx = canvas.getContext('2d')!;
    const bins = new Uint8Array(analyser.frequencyBinCount);
    let raf = 0;
    const draw = (): void => {
      raf = requestAnimationFrame(draw);
      analyser.getByteFrequencyData(bins);
      ctx.fillStyle = '#000'; ctx.fillRect(0, 0, canvas.width, canvas.height);
      const n = 32, step = Math.floor(bins.length / n), w = canvas.width / n;
      for (let i = 0; i < n; i++) {
        const v = bins[i * step] / 255, h = v * canvas.height;
        ctx.fillStyle = `rgb(${Math.round(80 + v * 175)},255,${Math.round(80 + v * 80)})`;
        ctx.fillRect(i * w, canvas.height - h, w - 1, h);
      }
    };
    draw();
    return () => cancelAnimationFrame(raf);
  }, [analyser, enabled]);
  return <canvas ref={ref} width={160} height={40} style={{ width: '100%', height: 40, background: '#000', imageRendering: 'pixelated' }} />;
}
```

```ts
// src/main/media/m3u.ts
import { isAbsolute, resolve, basename } from 'node:path';
export interface M3uItem { title: string; path?: string; url?: string }

export function parseM3u(text: string, baseDir: string): M3uItem[] {
  const out: M3uItem[] = [];
  let pendingTitle: string | null = null;
  for (const raw of text.split(/\r?\n/)) {
    const line = raw.trim();
    if (!line) continue;
    if (line.startsWith('#EXTINF:')) { pendingTitle = line.slice(line.indexOf(',') + 1).trim() || null; continue; }
    if (line.startsWith('#')) continue;
    if (/^https?:\/\//i.test(line)) out.push({ title: pendingTitle ?? line, url: line });
    else { const p = isAbsolute(line) ? line : resolve(baseDir, line); out.push({ title: pendingTitle ?? basename(p), path: p }); }
    pendingTitle = null;
  }
  return out;
}
export function toM3u(items: M3uItem[]): string {
  const lines = ['#EXTM3U'];
  for (const it of items) { lines.push(`#EXTINF:-1,${it.title}`); lines.push(it.url ?? it.path ?? ''); }
  return lines.join('\n') + '\n';
}
```

```ts
// src/renderer/modules/media/resolveSource.ts
export type Resolved = { kind: 'local' | 'stream'; src: string };
export function resolveSource(item: { path?: string; url?: string }, streamingEnabled: boolean): Resolved | null {
  if (item.path) return { kind: 'local', src: 'ga98media://track/' + encodeURIComponent(item.path) };
  if (item.url) { if (!streamingEnabled) return null; return { kind: 'stream', src: item.url }; }
  return null;
}
```
